===========================================
  Управление
===========================================

make build      # Собрать контейнеры
make up         # Запустить
make down       # Остановить
make restart    # Перезапустить
make logs       # Все логи
make logs-bot   # Логи бота
make logs-proxy # Логи прокси
make status     # Статус

# Или напрямую через docker compose:
docker compose build
docker compose up -d
docker compose down
docker compose restart
docker compose logs -f
docker compose ps

# Пересобрать только бота
./rebuild-bot.sh

===========================================
  Решение проблем
===========================================

# Пересоздать всё с нуля
make down
docker system prune -f
make build
make up

# Посмотреть ошибки
make logs | grep -i error

# Перезапустить только прокси
docker-compose restart proxy

# Перезапустить только бота
docker-compose restart bot

===========================================
  Открыть порты (если закрыты)

ufw status

# iptables
